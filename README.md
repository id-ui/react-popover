## Install

```bash
npm install --save @kaprisa57/react-popover
```

```bash
yarn add @kaprisa57/react-popover
```

```jsx
import React from 'react'

import Popover from '@kaprisa57/react-popover'

function Example() {
  return <Popover content="Hi!">
    <button>Open</button>
  </Popover>
}
```
